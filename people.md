<script>
  document.addEventListener('DOMContentLoaded', () => {
    const iconMarkup = {
      orcid: '<span class="person-link-text">iD</span>',
      website: '<svg viewBox="0 0 24 24" aria-hidden="true" focusable="false"><path d="M12 3.5a8.5 8.5 0 1 0 0 17 8.5 8.5 0 0 0 0-17Zm6.5 8h-3.2a14.6 14.6 0 0 0-1.2-4.6A6.56 6.56 0 0 1 18.5 11.5Zm-6.5 7.2c-.6 0-1.7-1.8-2.1-5.2h4.2c-.4 3.4-1.5 5.2-2.1 5.2Zm-2.3-7.2c.1-1.4.3-2.7.7-3.7.5-1.5 1.2-2.6 1.6-2.6s1.1 1.1 1.6 2.6c.3 1 .6 2.3.7 3.7H9.7Zm-4.2 0A6.56 6.56 0 0 1 9.9 6.9a14.6 14.6 0 0 0-1.2 4.6H5.5Zm0 1.6h3.2a14.6 14.6 0 0 0 1.2 4.6 6.56 6.56 0 0 1-4.4-4.6Zm8.6 4.6a14.6 14.6 0 0 0 1.2-4.6h3.2a6.56 6.56 0 0 1-4.4 4.6Z" fill="currentColor"/></svg>',
      linkedin: '<span class="person-link-text">in</span>',
      scholar: '<svg viewBox="0 0 24 24" aria-hidden="true" focusable="false"><path d="M12 4 2.5 9 12 14 21.5 9 12 4Zm-6.8 7.2V15c0 2.2 3 4 6.8 4s6.8-1.8 6.8-4v-3.8L12 15l-6.8-3.8Z" fill="currentColor"/></svg>'
    };

    document.querySelectorAll('.person-card').forEach((card) => {
      const name = card.querySelector('h2')?.textContent?.trim();
      const affiliation = card.querySelector('.person-affiliation')?.textContent?.trim() || '';
      const links = card.querySelector('.person-links');
      if (!name || !links) return;

      const baseQuery = encodeURIComponent(`${name} ${affiliation}`.trim());
      const scholarQuery = encodeURIComponent(name);

      const platforms = [
        { key: 'orcid', label: 'ORCID', href: `https://orcid.org/orcid-search/search?searchQuery=${baseQuery}` },
        { key: 'website', label: 'Website', href: `https://www.google.com/search?q=${baseQuery}` },
        { key: 'linkedin', label: 'LinkedIn', href: `https://www.linkedin.com/search/results/all/?keywords=${baseQuery}` },
        { key: 'scholar', label: 'Google Scholar', href: `https://scholar.google.com/scholar?q=${scholarQuery}` }
      ];

      platforms.forEach((platform) => {
        const anchor = document.createElement('a');
        anchor.className = 'person-link';
        anchor.href = platform.href;
        anchor.target = '_blank';
        anchor.rel = 'noopener';
        anchor.setAttribute('aria-label', `${platform.label} results for ${name}`);
        anchor.title = platform.label;
        anchor.innerHTML = iconMarkup[platform.key];
        links.appendChild(anchor);
      });
    });

    // ─── PEOPLE FILTER FUNCTIONALITY ───────────────────────────────
    const filterButtons = document.querySelectorAll('.filter-btn');
    const personCards = document.querySelectorAll('.person-card');
    let activeFilters = {
      university: '',
      department: ''
    };

    filterButtons.forEach((button) => {
      button.addEventListener('click', () => {
        const filterType = button.dataset.filter;
        const filterValue = button.dataset.value;

        // Remove active class from siblings
        document.querySelectorAll(`.filter-btn[data-filter="${filterType}"]`).forEach((btn) => {
          btn.classList.remove('filter-active');
        });

        // Add active class to clicked button
        button.classList.add('filter-active');

        // Update active filters
        activeFilters[filterType] = filterValue;

        // Filter person cards
        filterCards();
      });
    });

    const filterCards = () => {
      personCards.forEach((card) => {
        const affiliation = card.querySelector('.person-affiliation')?.textContent || '';
        
        // Check university filter
        let universityMatch = true;
        if (activeFilters.university) {
          universityMatch = affiliation.includes(activeFilters.university);
        }

        // Check department filter
        let departmentMatch = true;
        if (activeFilters.department) {
          departmentMatch = affiliation.includes(activeFilters.department);
        }

        // Show or hide card
        if (universityMatch && departmentMatch) {
          card.style.display = '';
          card.style.opacity = '1';
        } else {
          card.style.display = 'none';
          card.style.opacity = '0';
        }
      });
    };
  });
</script>
